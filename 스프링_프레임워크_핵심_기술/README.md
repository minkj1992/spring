arr = [6,2,15,18,23,14,14,11,5,8,13,6,15,17,13,25,15,18,18,13,6,7]

vals = [[],]
gid = 0
for i,v in enumerate(arr):
    if sum(vals[gid])+v > 65:
        gid+=1 
        vals.append([v,])
    else:
        vals[gid].append(v)

dates = [
'19.12.28',
'19.12.29',
'19.12.30',
'19.12.31',
'19.01.01']

i = 0
for val,date in zip(vals,dates):
    print(f'- {date}')
    for v in val:
        print(f'    - [ ] {i}강 {v}분')
        i+=1
